79void CMain::OnPaint(){
80 PAINTSTRUCT ps; //paint structure
81 HDC hdc = BeginPaint(m_hWnd, &ps); //device context
82 Gdiplus::Graphics graphics(hdc); //GDI+ graphics object
83 graphics.Clear(Gdiplus::Color::White); //clear to white
84
85 //get bitmap width and height
86
87 const int nBitmapWidth = m_pBitmap->GetWidth();
88 const int nBitmapHeight = m_pBitmap->GetHeight();
89
90 //get client rectangle
91
92 RECT rectClient; //for client rectangle
93 GetClientRect(m_hWnd, &rectClient); //get client rectangle
94 const int nClientWidth = rectClient.right - rectClient.left; //client width
95 const int nClientHeight = rectClient.bottom - rectClient.top; //client height
96
97 //compute destination rectangle
98
99 const int nDestSide = min(nClientWidth, nClientHeight); //dest width and ht
100
101 const int width = min(nDestSide, nBitmapWidth); //dest rect width
102 const int height = min(nDestSide, nBitmapHeight); //dest rect height
103
104 const int x = max(0, nClientWidth - width)/2; //x margin
105 const int y = max(0, nClientHeight - height)/2; //y margin
106
107 Gdiplus::Rect rectDest(x, y, width, height); //destination rectangle
108
109 //draw image to destination rectangle then clean up
110
111 graphics.DrawImage(m_pBitmap, rectDest);
112
113 EndPaint(m_hWnd, &ps); //this must be done last
114} //OnPaint

155void CMain::CreateTiles(){
156 Gdiplus::Image* pImage = Gdiplus::Bitmap::FromResource(m_hInstance,
157 MAKEINTRESOURCEW(IDB_BITMAP1));
158
159 const UINT w = pImage->GetWidth(); //tile width
160 const UINT h = pImage->GetHeight(); //tile height
161
162 for(UINT i=0; i<8; i++){
163 Gdiplus::Bitmap* pBitmap = new Gdiplus::Bitmap(w, h);
164 Gdiplus::Graphics graphics(pBitmap); //for drawing tiles in bitmap
165 graphics.DrawImage(pImage, 0, 0);
166
167 pBitmap->RotateFlip((Gdiplus::RotateFlipType)i); //rotate and flip
168
169 m_stdTile.push_back(pBitmap);
170 } //for
171} //CreateTiles

177UINT CMain::RandomTileIndex(){
178 const UINT nMax = (UINT)m_stdTile.size() - 1; //maximum tile index
179 return std::uniform_int_distribution<UINT>(0, nMax)(*m_pRand);

205void CMain::Generate(const eTesselation tesselation){
206 static const UINT nOriginal[8][8]{ //tile indices in original image
207 {0, 1, 5, 4, 5, 6, 0, 3},
208 {1, 2, 6, 5, 6, 7, 3, 2},
209 {0, 1, 0, 1, 0, 1, 0, 1},
210 {1, 2, 3, 0, 3, 0, 3, 0},
211 {5, 6, 0, 3, 0, 1, 5, 4},
212 {6, 7, 3, 2, 1, 2, 6, 5},
213 {0, 1, 0, 1, 0, 1, 0, 1},
214 {3, 0, 3, 0, 1, 2, 3, 0}
215 }; //nOriginal
216
217 CheckMenuItems(tesselation);
218
219 //generate tiled image
220
221 const UINT w = m_stdTile[0]->GetWidth(); //tile width
222 const UINT h = m_stdTile[0]->GetHeight(); //tile height
223
224 const UINT nSize = 8; //number of tiles per side of bitmap
225
226 CreateBitmap(w*nSize, h*nSize); //create bitmap
227 Gdiplus::Graphics graphics(m_pBitmap); //for drawing tiles in bitmap
228 graphics.Clear(Gdiplus::Color(143, 158, 104)); //fill with green
229
230 int y = 0; //vertical tile coordinate
231
232 for(UINT i=0; i<nSize; i++){ //vertical tile index
233 int x = 0; //horizontal tile coordinate
234
235 for(UINT j=0; j<nSize; j++){ //horizontal tile index
236 UINT nTileIndex = 0; //tile index for current tile
237
238 switch(tesselation){
239 case eTesselation::Original: nTileIndex = nOriginal[i][j]; break;
240 case eTesselation::Random: nTileIndex = RandomTileIndex(); break;
241 } //switch
242
243 graphics.DrawImage(m_stdTile[nTileIndex], x, y); //draw tile
244 x += w; //move right to next tile
245 } //for
246
247 y += h; //move down to next tile
248 } //for
249} //Generate
