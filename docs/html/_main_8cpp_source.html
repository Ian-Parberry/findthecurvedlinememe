46LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam){
47 UINT nMenuId = 0; //menu identifier for menu command messages
48
49 switch(message){
50 case WM_CREATE: //window has been created
51 g_pMain = new CMain(g_hInstance, hWnd); //create the main class
52 return 0;
53
54 case WM_DESTROY: //window has been removed from the screen
55 delete g_pMain; //delete the main class
56 PostQuitMessage(0); //ready to shut down
57 return 0;
58
59 case WM_PAINT: //window needs to be redrawn
60 g_pMain->OnPaint();
61 return 0;
62
63 case WM_COMMAND: //user has selected a command from the menu
64 nMenuId = LOWORD(wParam); //menu id
65
66 switch(nMenuId){
67 case IDM_FILE_SAVE:
68 SaveBitmap(hWnd, L"Output", g_pMain->GetBitmap());
69 break;
70
71 case IDM_FILE_QUIT: //so long, farewell, auf weidersehn, goodbye!
72 SendMessage(hWnd, WM_CLOSE, 0, 0);
73 break;
74
75 case IDM_TESS_ORIGINAL: //original tesselation
76 g_pMain->Generate(CMain::eTesselation::Original);
77 InvalidateRect(hWnd, nullptr, FALSE);
78 break;
79
80 case IDM_TESS_RANDOM: //random tesselation
81 g_pMain->Generate(CMain::eTesselation::Random);
82 InvalidateRect(hWnd, nullptr, FALSE);
83 break;
84
85 case IDM_HELP_HELP: //show help
86 ShellExecute(0, 0,
87 "https://ian-parberry.github.io/findthecurvedlinememe/html",
88 0, 0, SW_SHOW);
89 break;
90
91 case IDM_HELP_ABOUT: //show ABout dialog box.
92 MessageBox(nullptr,
93 "Copyright Â© Ian Parberry, 2022.\nSource code available under the MIT License from https://github.com/Ian-Parberry/.",
94 "About", MB_ICONINFORMATION | MB_OK);
95 break;
96 } //switch
97
98 return 0; //all is good
99
100 default:
101 return DefWindowProc(hWnd, message, wParam, lParam); //not my message
102 } //switch
103} //WndProc
