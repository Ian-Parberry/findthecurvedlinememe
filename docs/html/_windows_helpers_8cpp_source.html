40std::wstring FileNameBase(const std::wstring& wstrPath){
41 const size_t n0 = wstrPath.find_last_of(L"\\");
42
43 std::wstring wstrResult = (n0 == std::string::npos)? wstrPath:
44 wstrPath.substr(n0 + 1, wstrPath.size() - n0);
45
46 const size_t n1 = wstrResult.find_last_of(L".");
47
48 if(n1 != std::string::npos)
49 wstrResult = wstrResult.substr(0, n1);
50
51 return wstrResult;
52} //FileNameBase

126HRESULT GetEncoderClsid(const WCHAR* format, CLSID* pClsid){
127 UINT num = 0; //number of image encoders
128 UINT n = 0; //size of the image encoder array in bytes
129 HRESULT hr = E_FAIL; //return result
130
131 Gdiplus::ImageCodecInfo* pCodecInfo = nullptr; //for codec info
132 if(FAILED(Gdiplus::GetImageEncodersSize(&num, &n)))return E_FAIL; //get sizes
133 if(n == 0)return E_FAIL; //there are no encoders
134
135 pCodecInfo = (Gdiplus::ImageCodecInfo*)(malloc(n)); //allocate codec info memory
136 if(pCodecInfo == nullptr)return E_FAIL; //malloc failed (as if)
137 if(FAILED(GetImageEncoders(num, n, pCodecInfo)))return E_FAIL; //get encoders
138
139 for(UINT j=0; j<num && hr!=S_OK; j++) //for each encoder, while not found
140 if(wcscmp(pCodecInfo[j].MimeType, format) == 0){ //found the codex we want
141 *pClsid = pCodecInfo[j].Clsid; //return it
142 hr = S_OK; //success
143 } //if
144
145 free(pCodecInfo); //clean up
146 return hr;
147} //GetEncoderClsid

161HRESULT SaveBitmap(HWND hwnd, const std::wstring& wstrName,
162 Gdiplus::Bitmap* pBitmap)
163{
164 COMDLG_FILTERSPEC filetypes[] = { //png files only
165 {L"PNG Files", L"*.png"}
166 }; //filetypes
167
168 std::wstring wstrFileName; //result
169 CComPtr<IFileSaveDialog> pDlg; //pointer to save dialog box
170 CComPtr<IShellItem> pItem; //item pointer
171 LPWSTR pwsz = nullptr; //pointer to null-terminated wide string for result
172
173 //fire up the save dialog box
174
175 if(FAILED(pDlg.CoCreateInstance(__uuidof(FileSaveDialog))))return E_FAIL;
176
177 pDlg->SetFileTypes(_countof(filetypes), filetypes); //set file types to png
178 pDlg->SetTitle(L"Save Image"); //set title bar text
179 pDlg->SetFileName(wstrName.c_str()); //set default file name
180 pDlg->SetDefaultExtension(L"png"); //set default extension
181
182 if(FAILED(pDlg->Show(hwnd)))return E_FAIL; //show the dialog box
183 if(FAILED(pDlg->GetResult(&pItem)))return E_FAIL; //get the result item
184 if(FAILED(pItem->GetDisplayName(SIGDN_FILESYSPATH, &pwsz)))return E_FAIL; //get file name
185
186 //wstrFileName should now contain the selected file name
187
188 CLSID clsid; //for PNG class id
189 if(FAILED(GetEncoderClsid((WCHAR*)L"image/png", &clsid)))return E_FAIL; //get
190 pBitmap->Save(pwsz, &clsid, nullptr); //the actual save happens here
191
192 CoTaskMemFree(pwsz); //clean up
193
194 return S_OK;
195} //SaveBitmap
